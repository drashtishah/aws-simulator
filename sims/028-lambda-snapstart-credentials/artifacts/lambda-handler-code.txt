package com.tidewaterpay.payments;

import com.amazonaws.services.lambda.runtime.Context;
import com.amazonaws.services.lambda.runtime.RequestHandler;
import software.amazon.awssdk.services.dynamodb.DynamoDbClient;
import software.amazon.awssdk.services.dynamodb.model.PutItemRequest;
import software.amazon.awssdk.services.secretsmanager.SecretsManagerClient;
import software.amazon.awssdk.services.secretsmanager.model.GetSecretValueRequest;
import java.util.Map;

public class Handler implements RequestHandler<Map<String, Object>, Map<String, Object>> {

    // PROBLEM: these SDK clients are constructed at class-load time, before
    // the SnapStart checkpoint. Their credential provider caches are frozen
    // into the snapshot and reused across every cold start for as long as
    // the published version lives.
    private static final DynamoDbClient dynamo = DynamoDbClient.create();
    private static final SecretsManagerClient secrets = SecretsManagerClient.create();

    // PROBLEM: the signing key is also fetched at init time and cached in a
    // static field. If the secret rotates between snapshot and restore, the
    // function keeps using the old key.
    private static final String SIGNING_KEY = secrets
        .getSecretValue(GetSecretValueRequest.builder()
            .secretId(System.getenv("SIGNING_KEY_SECRET_ID"))
            .build())
        .secretString();

    @Override
    public Map<String, Object> handleRequest(Map<String, Object> event, Context ctx) {
        // ... validation, parsing ...

        // This PutItem call uses the DynamoDB client whose credentials are
        // in the snapshot. On cold-started restores past expiry it throws
        // software.amazon.awssdk.services.dynamodb.model.DynamoDbException
        // with a cause of software.amazon.awssdk.awscore.exception.AwsServiceException
        // message "ExpiredTokenException: The security token included in the request is expired"
        dynamo.putItem(PutItemRequest.builder()
            .tableName(System.getenv("PAYMENTS_TABLE"))
            .item(toItem(event))
            .build());

        return Map.of("status", "ok");
    }

    private Map<String, software.amazon.awssdk.services.dynamodb.model.AttributeValue> toItem(Map<String, Object> event) {
        // ... marshal ...
        return Map.of();
    }
}
