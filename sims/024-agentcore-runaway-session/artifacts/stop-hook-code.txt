from agentcore_sdk import StopDecision

def evaluate(tool_output: bytes, scratchpad: list) -> StopDecision:
    # Decode the latest tool output to a string so the rules engine can scan it.
    text = tool_output.decode()  # PROBLEM: no errors='replace', raises on non-UTF8 bytes

    # Short-circuit: if the tool already returned a final answer, stop.
    if "<final-answer>" in text:
        return StopDecision(stop=True, reason="tool returned final-answer tag")

    # Heuristic: if the last three turns all called the same tool, stop.
    last_three = scratchpad[-3:]
    if len(last_three) == 3 and all(t.tool == last_three[0].tool for t in last_three):
        return StopDecision(stop=True, reason="same tool three turns in a row")

    return StopDecision(stop=False, reason="continue")
